# pyone_test
Testing OpenNebula API with pyone library




Installation:

# Install venv
sudo apt-get install python3-venv

# create working dir
mkdir ./one_api && cd one_api

# in working dir
python3 -m venv .venv

# activate env
. .venv/bin/activate

# Install libraries
pip3 install lxml==4.4.0
pip3 install pyone
